DELIMITER $$
CREATE DEFINER = 'root'@'localhost'
PROCEDURE view_itis_tns_long()
BEGIN
SELECT
itis.taxonomic_units.n_usage,
itis.taxonomic_units.credibility_rtng,
itis.taxonomic_units.unaccept_reason, 
itis.taxon_unit_types.rank_name,   
itis.taxonomic_units.complete_name,  
itis.strippedauthor.shortauthor,
`level`,
    @tsnCount := 1 + LENGTH(itis.hierarchy.hierarchy_string) - LENGTH(REPLACE(itis.hierarchy.hierarchy_string, '-', '')) AS tnsCount,
    hierarchy_string,
    hierarchy.TSN,
    IF(@tsnCount > 0, SUBSTRING_INDEX(SUBSTRING_INDEX(itis.hierarchy.hierarchy_string, '-', 1), '-', -1), '') AS TSN_01,
    IF(@tsnCount > 1, SUBSTRING_INDEX(SUBSTRING_INDEX(itis.hierarchy.hierarchy_string, '-', 2), '-', -1), '') AS TSN_02,
    IF(@tsnCount > 2, SUBSTRING_INDEX(SUBSTRING_INDEX(itis.hierarchy.hierarchy_string, '-', 3), '-', -1), '') AS TSN_03,
    IF(@tsnCount > 3, SUBSTRING_INDEX(SUBSTRING_INDEX(itis.hierarchy.hierarchy_string, '-', 4), '-', -1), '') AS TSN_04,
    IF(@tsnCount > 4, SUBSTRING_INDEX(SUBSTRING_INDEX(itis.hierarchy.hierarchy_string, '-', 5), '-', -1), '') AS TSN_05,
    IF(@tsnCount > 5, SUBSTRING_INDEX(SUBSTRING_INDEX(itis.hierarchy.hierarchy_string, '-', 6), '-', -1), '') AS TSN_06,
    IF(@tsnCount > 6, SUBSTRING_INDEX(SUBSTRING_INDEX(itis.hierarchy.hierarchy_string, '-', 7), '-', -1), '') AS TSN_07,
    IF(@tsnCount > 7, SUBSTRING_INDEX(SUBSTRING_INDEX(itis.hierarchy.hierarchy_string, '-', 8), '-', -1), '') AS TSN_08,
    IF(@tsnCount > 8, SUBSTRING_INDEX(SUBSTRING_INDEX(itis.hierarchy.hierarchy_string, '-', 9), '-', -1), '') AS TSN_09,
    IF(@tsnCount > 9, SUBSTRING_INDEX(SUBSTRING_INDEX(itis.hierarchy.hierarchy_string, '-', 10), '-', -1), '') AS TSN_10,
    IF(@tsnCount > 10, SUBSTRING_INDEX(SUBSTRING_INDEX(itis.hierarchy.hierarchy_string, '-', 11), '-', -1), '') AS TSN_11,
    IF(@tsnCount > 11, SUBSTRING_INDEX(SUBSTRING_INDEX(itis.hierarchy.hierarchy_string, '-', 12), '-', -1), '') AS TSN_12,
    IF(@tsnCount > 12, SUBSTRING_INDEX(SUBSTRING_INDEX(itis.hierarchy.hierarchy_string, '-', 13), '-', -1), '') AS TSN_13,
    IF(@tsnCount > 13, SUBSTRING_INDEX(SUBSTRING_INDEX(itis.hierarchy.hierarchy_string, '-', 14), '-', -1), '') AS TSN_14,
    IF(@tsnCount > 14, SUBSTRING_INDEX(SUBSTRING_INDEX(itis.hierarchy.hierarchy_string, '-', 15), '-', -1), '') AS TSN_15,
    IF(@tsnCount > 15, SUBSTRING_INDEX(SUBSTRING_INDEX(itis.hierarchy.hierarchy_string, '-', 16), '-', -1), '') AS TSN_16,
    IF(@tsnCount > 16, SUBSTRING_INDEX(SUBSTRING_INDEX(itis.hierarchy.hierarchy_string, '-', 17), '-', -1), '') AS TSN_17,
itis.taxonomic_units.taxon_author_id,
itis.taxonomic_units.kingdom_id,
itis.taxonomic_units.rank_id
FROM itis.taxonomic_units
LEFT JOIN itis.hierarchy ON itis.hierarchy.TSN=itis.taxonomic_units.tsn
LEFT JOIN itis.strippedauthor ON itis.taxonomic_units.taxon_author_id=itis.strippedauthor.taxon_author_id
LEFT JOIN itis.taxon_unit_types 
ON   itis.taxonomic_units.kingdom_id=itis.taxon_unit_types.kingdom_id 
AND  itis.taxonomic_units.rank_id=itis.taxon_unit_types.rank_id
WHERE itis.hierarchy.hierarchy_string LIKE "%-948936%"   AND itis.taxonomic_units.n_usage='valid'
;
END
$$
DELIMITER ;
