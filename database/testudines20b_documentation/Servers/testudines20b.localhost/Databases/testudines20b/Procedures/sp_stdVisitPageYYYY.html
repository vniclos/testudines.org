DELIMITER $$
CREATE DEFINER = 'root'@'localhost'
PROCEDURE sp_stdVisitPageYYYY(IN iYYYY INT)
BEGIN
SELECT stdUrl, COUNT(*) FROM tstd_vistitlog  
where stdYear = iYYYY 
GROUP BY stdUrl order by  COUNT(*) Desc;
END
$$
DELIMITER ;
