CREATE 
DEFINER = 'root'@'localhost'
VIEW view_itis_tsn
AS
SELECT
  `itis`.`hierarchy`.`hierarchy_string` AS `hierarchy_string`,
  `itis`.`hierarchy`.`TSN` AS `TSN`,
  `itis`.`hierarchy`.`Parent_TSN` AS `Parent_TSN`,
  `itis`.`taxonomic_units`.`complete_name` AS `complete_name`,
  `itis`.`taxonomic_units`.`taxon_author_id` AS `taxon_author_id`,
  `itis`.`strippedauthor`.`shortauthor` AS `shortauthor`,
  `itis`.`taxonomic_units`.`credibility_rtng` AS `credibility_rtng`,
  `itis`.`taxonomic_units`.`unaccept_reason` AS `unaccept_reason`,
  `itis`.`taxonomic_units`.`name_usage` AS `name_usage`
FROM ((`itis`.`hierarchy`
  LEFT JOIN `itis`.`taxonomic_units`
    ON (`itis`.`hierarchy`.`TSN` = `itis`.`taxonomic_units`.`tsn`))
  LEFT JOIN `itis`.`strippedauthor`
    ON (`itis`.`taxonomic_units`.`taxon_author_id` = `itis`.`strippedauthor`.`taxon_author_id`))
WHERE `itis`.`hierarchy`.`hierarchy_string` LIKE '%-948936%';
