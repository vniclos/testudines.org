CREATE TABLE tdoclng_media (
  DocId BIGINT(20) UNSIGNED NOT NULL,
  DocLngId CHAR(2) NOT NULL DEFAULT 'en' COMMENT 'ID Root, for document of database(tdoc.docid)',
  UID VARCHAR(64) NOT NULL DEFAULT '' COMMENT 'Universal ID for not crash with other files ',
  URL VARCHAR(450) NOT NULL DEFAULT '' COMMENT 'Url ful path 260',
  FileNameOnSource VARCHAR(250) NOT NULL DEFAULT '' COMMENT 'Original filename on source',
  FileNameShort VARCHAR(250) NOT NULL DEFAULT '' COMMENT 'Sort filename',
  Type VARCHAR(25) NOT NULL DEFAULT '' COMMENT 'imgage, video, ... (Not in use at this moment)',
  ImgWith INT(5) UNSIGNED NOT NULL DEFAULT 0,
  imgHeight INT(5) UNSIGNED NOT NULL DEFAULT 0,
  IsGalleryVisible TINYINT(1) NOT NULL DEFAULT 1 COMMENT 'If is visible in gallery of web site',
  IsDownloadable TINYINT(1) NOT NULL DEFAULT 1 COMMENT 'If is Donloadable',
  IsVisible TINYINT(1) NOT NULL DEFAULT 1,
  Title VARCHAR(250) NOT NULL DEFAULT '' COMMENT 'Title',
  Author VARCHAR(80) NOT NULL DEFAULT '' COMMENT 'Cite Author',
  Source VARCHAR(80) NOT NULL DEFAULT '' COMMENT 'Cite Source',
  Notes MEDIUMTEXT NOT NULL COMMENT 'Notes',
  CreateYear INT(4) NOT NULL,
  CreateGeoLocation VARCHAR(80) NOT NULL DEFAULT '',
  CreateGeoGPS VARCHAR(250) NOT NULL DEFAULT '',
  CreateDate DATETIME NOT NULL,
  LogFromIp VARCHAR(50) NOT NULL DEFAULT '',
  LogFromDate DATE NOT NULL,
  PRIMARY KEY (DocId, DocLngId)
)
ENGINE = INNODB,
CHARACTER SET utf8,
COLLATE utf8_general_ci,
COMMENT = 'List of images in database, ',
ROW_FORMAT = Dynamic,
TRANSACTIONAL = 0;
ALTER TABLE tdoclng_media 
  ADD INDEX FileNameURL(URL(255));
