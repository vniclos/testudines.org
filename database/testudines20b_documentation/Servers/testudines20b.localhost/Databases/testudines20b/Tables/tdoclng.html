CREATE TABLE tdoclng (
  DocId BIGINT(20) UNSIGNED NOT NULL COMMENT 'ID Root, for document of database(tdoc.docid)',
  DocLngId CHAR(2) NOT NULL COMMENT 'ID Root Laguage, for document of database(tdoclng.doclngid)',
  DocLngUrlId VARCHAR(250) NOT NULL DEFAULT '' COMMENT 'Url for this laguage',
  DocLngUrlTitle VARCHAR(250) NOT NULL DEFAULT '' COMMENT 'Title for this language',
  DocId_tdoclng_Todo INT(11) NOT NULL DEFAULT 0 COMMENT 'List of things pending todo, ',
  DocLngUrlIdChk TINYINT(1) NOT NULL DEFAULT 0 COMMENT 'true if is verified low',
  DocLngUrlTitleChk TINYINT(1) NOT NULL DEFAULT 0 COMMENT 'True if cheched ',
  DocLngMetaTitle VARCHAR(250) NOT NULL DEFAULT '' COMMENT 'Meta title',
  DocLngMetaDescription VARCHAR(250) NOT NULL DEFAULT '',
  DocLngMetaKeyWords VARCHAR(100) NOT NULL DEFAULT '',
  DocLngMetaLanguage VARCHAR(3) NOT NULL DEFAULT '',
  DocLngRevisit_after VARCHAR(10) NOT NULL DEFAULT '',
  DocLngMetaAuthor MEDIUMTEXT NOT NULL,
  DocLngMetaTranslators VARCHAR(150) NOT NULL DEFAULT '',
  DocLngMetaRobots VARCHAR(35) NOT NULL DEFAULT '',
  DocLngMetaContentType VARCHAR(35) NOT NULL DEFAULT '',
  DocLngDateUpdate DATETIME NOT NULL COMMENT 'Date update this translation',
  DocLngDateCreation DATETIME NOT NULL,
  DocLngStatusRevision VARCHAR(50) NOT NULL DEFAULT '' COMMENT 'if is fine revisel',
  DocLngNotes MEDIUMTEXT NOT NULL COMMENT 'Notes for this translation document',
  PRIMARY KEY (DocId, DocLngId)
)
ENGINE = INNODB,
CHARACTER SET utf8,
COLLATE utf8_general_ci,
COMMENT = 
'Root doc id for especific language ',

TRANSACTIONAL = 0;
