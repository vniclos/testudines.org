CREATE TABLE tmst_avatars (
  AvatarId BIGINT(20) UNSIGNED NOT NULL AUTO_INCREMENT,
  Description VARCHAR(150) NOT NULL DEFAULT '',
  FileNameSrc VARCHAR(50) NOT NULL DEFAULT '',
  IsActive TINYINT(1) NOT NULL DEFAULT 1,
  IsPublicUse TINYINT(1) NOT NULL DEFAULT 1,
  UploadIdUser MEDIUMINT(8) UNSIGNED NOT NULL DEFAULT 0,
  UploadDate TIMESTAMP NOT NULL DEFAULT current_timestamp() ON UPDATE CURRENT_TIMESTAMP,
  UploadIP VARCHAR(50) NOT NULL DEFAULT '',
  PRIMARY KEY (AvatarId)
)
ENGINE = INNODB,
CHARACTER SET utf8,
COLLATE utf8_general_ci,
COMMENT = 'Master Not in use avatars for users',
ROW_FORMAT = Dynamic,
TRANSACTIONAL = 0;
