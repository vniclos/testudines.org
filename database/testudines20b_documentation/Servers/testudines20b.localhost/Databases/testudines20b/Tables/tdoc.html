CREATE TABLE tdoc (
  DocId BIGINT(20) UNSIGNED NOT NULL AUTO_INCREMENT COMMENT 'ID Root, for document of database',
  DocLngId_Main CHAR(2) NOT NULL DEFAULT 'en' COMMENT 'ID Root Language, for al database',
  DocTypeId CHAR(50) NOT NULL DEFAULT '' COMMENT 'Type document, Article, Taxon, Notice.....',
  DocTypeGroup VARCHAR(50) NOT NULL DEFAULT '' COMMENT 'Group of documents',
  DoctypeGroupSub CHAR(255) NOT NULL DEFAULT '' COMMENT 'Sub type of document',
  DocIsTraslatable TINYINT(1) NOT NULL DEFAULT 1 COMMENT 'true=is traslatable',
  DocImgThumbnail VARCHAR(250) NOT NULL DEFAULT '' COMMENT 'Url thumbnail image',
  DocDocUploaded VARCHAR(250) NOT NULL DEFAULT '' COMMENT 'If is updaded show source url',
  DocIsEditable TINYINT(1) NOT NULL DEFAULT 1 COMMENT 'true = Is Edictable',
  DocIsActive TINYINT(1) NOT NULL DEFAULT 1 COMMENT 'true=visible on web site',
  DocUserId_Creator MEDIUMINT(45) NOT NULL DEFAULT 0 COMMENT 'Id Number creator',
  DocAuthor VARCHAR(80) NOT NULL DEFAULT '' COMMENT 'Cite Author',
  DocAuthors VARCHAR(150) NOT NULL DEFAULT '' COMMENT 'Cite Autors et All',
  DocBibliography LONGTEXT NOT NULL COMMENT 'Bibliography',
  DocAcknowledgements LONGTEXT NOT NULL COMMENT 'List acknowlegmts',
  DocAuthorizations LONGTEXT DEFAULT '' COMMENT 'List of especial authoriztion (Images, media, ...)',
  DocNotes LONGTEXT NOT NULL COMMENT 'Notes for other thins',
  DocDateCreation DATETIME NOT NULL COMMENT 'Date creation of document',
  DocDateUpdate TIMESTAMP NOT NULL DEFAULT current_timestamp() ON UPDATE CURRENT_TIMESTAMP COMMENT 'Date Update',
  DocStdVisitLast DATETIME NOT NULL COMMENT 'Date of las visit on web site',
  DocStdVisitCount MEDIUMINT(9) NOT NULL DEFAULT 0 COMMENT 'Counter of visit of web site',
  DocStdValLow MEDIUMINT(9) NOT NULL DEFAULT 0 COMMENT 'Valoration Low',
  DocStdValMed MEDIUMINT(9) NOT NULL DEFAULT 0 COMMENT 'Valoration med',
  DocStdValHig MEDIUMINT(9) NOT NULL DEFAULT 0 COMMENT 'Valoration high',
  DocCountriesCode CHAR(150) NOT NULL DEFAULT '' COMMENT 'Countries code where is present',
  DocSiteMapPriority CHAR(4) NOT NULL DEFAULT '0.6' COMMENT 'Site map Crawler use',
  DocSiteMapChangefreq VARCHAR(10) NOT NULL DEFAULT 'month' COMMENT 'Site map Crawler use',
  DocPdfUrl VARCHAR(250) NOT NULL DEFAULT '''''' COMMENT 'Pdf url for download (If exist)',
  DocPdfVersion BIGINT(20) NOT NULL DEFAULT 0 COMMENT 'Pdf Counter of changes',
  DocPdfTitle VARCHAR(250) NOT NULL DEFAULT '''''' COMMENT 'Pdf title',
  PRIMARY KEY (DocId)
)
ENGINE = INNODB,
CHARACTER SET utf8,
COLLATE utf8_general_ci,
COMMENT = 
'Main table of databse root for all documents, is same for all laguajes and documents like Taxa, articles, media, .....',
ROW_FORMAT = Dynamic,
TRANSACTIONAL = 0;
