CREATE TABLE tmst_geo_countrylanguage (
  CountryCode CHAR(3) NOT NULL DEFAULT '',
  LanguageName CHAR(30) NOT NULL DEFAULT '',
  IsOfficial ENUM('T','F') NOT NULL DEFAULT 'F',
  Percentage FLOAT(4, 1) NOT NULL DEFAULT 0.0,
  IsActive TINYINT(1) DEFAULT 1,
  PRIMARY KEY (CountryCode, LanguageName)
)
ENGINE = MYISAM,
CHARACTER SET utf8,
CHECKSUM = 0,
COLLATE utf8_general_ci,
COMMENT = 'Master of countries language',
ROW_FORMAT = Fixed,
TRANSACTIONAL = 0;
