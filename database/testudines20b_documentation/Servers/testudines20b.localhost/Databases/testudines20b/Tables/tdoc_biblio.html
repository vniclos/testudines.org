CREATE TABLE tdoc_biblio (
  DocId BIGINT(20) UNSIGNED NOT NULL COMMENT 'ID Root, for document of database(tdoc.docid)',
  CiteAutorYearABC VARCHAR(250) NOT NULL DEFAULT '',
  CiteHtmlToolTip MEDIUMTEXT DEFAULT NULL,
  CiteHtmlFull MEDIUMTEXT DEFAULT NULL,
  Authors TEXT DEFAULT NULL,
  Title VARCHAR(250) NOT NULL DEFAULT '',
  PublicationType VARCHAR(50) NOT NULL DEFAULT '',
  PublicationName VARCHAR(250) NOT NULL DEFAULT '',
  PublicationDatePage VARCHAR(50) NOT NULL DEFAULT '',
  PublicationNotes MEDIUMTEXT DEFAULT NULL,
  PublicationReaded VARCHAR(10) NOT NULL,
  URPublication TEXT DEFAULT NULL,
  URLocalDownload TEXT DEFAULT NULL
)
ENGINE = INNODB,
CHARACTER SET utf8,
COLLATE utf8_general_ci,
COMMENT = 'Bibliography list',
ROW_FORMAT = Dynamic,
TRANSACTIONAL = 0;
ALTER TABLE tdoc_biblio
