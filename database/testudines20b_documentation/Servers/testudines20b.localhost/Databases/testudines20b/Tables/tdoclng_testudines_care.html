CREATE TABLE tdoclng_testudines_care (
  DocId BIGINT(20) UNSIGNED NOT NULL COMMENT 'ID Root, for document of database(tdoc.docid)',
  DocLngId CHAR(2) NOT NULL COMMENT 'ID Root Laguage, for document of database(tdoclng.doclngid)',
  LCareAbstract TEXT DEFAULT '' COMMENT 'Abstract short care, acuatic, terrestrial, temperature footd',
  LCareValues TEXT DEFAULT '' COMMENT 'Care values',
  LCareVivariumIndoor TEXT DEFAULT '' COMMENT 'Vivarium indoor',
  LCareVivariumOutdoor TEXT DEFAULT '' COMMENT 'Vivarium outdoor',
  LCareFood TEXT DEFAULT '' COMMENT 'Foods',
  LCareBreeding TEXT DEFAULT '' COMMENT 'Breading',
  LCareNotes TEXT DEFAULT '' COMMENT 'Notes',
  LCareEspecialist TEXT DEFAULT '' COMMENT 'Especialist for consults',
  PRIMARY KEY (DocId, DocLngId)
)
ENGINE = INNODB,
CHARACTER SET utf8,
COLLATE utf8_general_ci,
COMMENT = 'Testudines taxon id abstract. Chapter Care sheet',
ROW_FORMAT = Dynamic,
TRANSACTIONAL = 0;
