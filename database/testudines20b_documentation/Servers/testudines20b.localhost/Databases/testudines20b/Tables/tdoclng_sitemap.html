CREATE TABLE tdoclng_sitemap (
  DocId BIGINT(20) UNSIGNED NOT NULL COMMENT 'ID Root, for document of database(tdoc.docid)',
  DocLngId VARCHAR(2) NOT NULL DEFAULT '' COMMENT 'ID Root Laguage, for document of database(tdoclng.doclngid)',
  DocCapituleId VARCHAR(32) NOT NULL DEFAULT '' COMMENT 'Blanco o capitulo del documento',
  loc_docid VARCHAR(300) NOT NULL DEFAULT '/',
  loc_title VARCHAR(300) NOT NULL DEFAULT '/',
  changefreq VARCHAR(20) NOT NULL DEFAULT 'monthly',
  lastmod VARCHAR(10) NOT NULL DEFAULT '',
  priority VARCHAR(10) NOT NULL DEFAULT '',
  LastUpdate TIMESTAMP NOT NULL DEFAULT current_timestamp() ON UPDATE CURRENT_TIMESTAMP,
  urltemplate VARCHAR(300) NOT NULL DEFAULT '',
  DocClass VARCHAR(30) NOT NULL DEFAULT '',
  PRIMARY KEY (DocId, DocLngId, DocCapituleId)
)
ENGINE = INNODB,
CHARACTER SET utf8,
COLLATE utf8_general_ci,
COMMENT = 'Site mapp tool for add to searchers',
ROW_FORMAT = Dynamic,
TRANSACTIONAL = 0;
